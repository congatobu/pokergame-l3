238 /**
239  * Affiche la liste des parties.
240  * @author Paul	Mura
241  * @param {String} message Liste de partie sous la forme: Partie1/nombre_joueurs/nombre_max@Partie2/nombre_joueurs/nombre_max...
242  */
243 function affiche_liste(message){
244 	var html='';
245 	var liste = message.split("@");
246 	var partie;
247 	
248 	html += "<TR id=\'titretable\'>"+$("#titretable").html()+"</TR>";
249 	for(i=1;i<liste.length;i++){
250 		partie = liste[i].split("/");
251 		html +='<TR><TD class=\'ligne_partie\' align=\'center\'><a style=\'display : block; text-decoration : none;\' href=\"javascript:rejoindre_partie(\''+partie[0]+'\');\">'+partie[0]+'</a></TD>'
252 		for(j=1;j<partie.length;j++){
253 			html +='<TD class=\'ligne_partie\' align=\'center\'>'+partie[j]+'</TD>';
254 		}
255 		html +='</TR>';
256 	}
257 	
258 	$("#tableparties").text("");
259 	$("#tableparties").append(html);
260 	$("#tableau").show("slow");
261 }

367 /**
368  * Détecte les changement dee créateur de partie et appelle la fonction partie.miseajourJoueurs(String[]).
369  * @author Paul	Mura
370  * @param {String[]} liste Tableau contenant la liste des joueurs de la partie.
371  */
372 function lister_joueurs(liste){
373 	for(i=1;i<liste.length;i++)
374 		if(liste[i].substring(0,1)=="$"){
375 			liste[i]=liste[i].substring(1,liste[i].length);
376 			if(!encours && liste[i]==pseudo){
377 				$("#commencer").show("slow"); 
378 				if(!createur){
379 					alert("Le créateur a quitté la partie.\nVous devenez le nouveau créateur!");
380 					createur = true;
381 				}
382 			}
383 		}
384 	partie.miseajourJoueurs(liste.slice(1,liste.length));
385 }
