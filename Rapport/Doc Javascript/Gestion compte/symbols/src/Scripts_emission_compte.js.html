 25 /**
 26  * Envoi une demande de changement de pseudo au serveur, après avoir vérifié la validité des variables.
 27  * @author Paul	Mura
 28  * @param {HTMLObject} pseudo L'ancien pseudo de l'utilisateur.
 29  * @param {HTMLObject} nouveau Le nouveau pseudo de l'utilisateur.
 30  * @param {HTMLObject} pass Le mot de passe de l'utilisateur.
 31  */
 32 function changer_psd(pseudo, nouveau, pass){
 33 	resetpseudo();
 34 	if (pseudo.value==''){
 35 		$("#labpseudochmt").text("/!\\ entrez votre pseudo.");
 36 		nouveau.focus();
 37 	}
 38 	else if (nouveau.value==''){
 39 		$("#labnouvpseudo").text("/!\\ entrez le nouveau pseudo.");
 40 		nouveau.focus();
 41 	}
 42 	else if (pass.value==''){
 43 		$("#labpasschmtpsd").text("/!\\ entrez votre mot de passe.");
 44 		pass.focus();
 45 	}
 46 	else{
 47 		set_reception("EM_CMPT", "ACTPSEUDO");
 48 		connexion_serveur();
 49 		socket_send("ACTPSEUDO@"+pseudo.value+"@"+pass.value+"@"+nouveau.value);
 50 	}
 51 }

 53 /**
 54  * Envoi une demande de changement de mot de passe au serveur, après avoir vérifié la validité des variables.
 55  * @author Paul	Mura
 56  * @param {HTMLObject} pseudo Le pseudo de l'utilisateur.
 57  * @param {HTMLObject} ancien Son ancien mot de passe.
 58  * @param {HTMLObject} nouveau Son nouveau mot de passe.
 59  * @param {HTMLObject} nouveau2 La confirmation du mot de passe.
 60  */
 61 function changer_mdp(pseudo, ancien, nouveau, nouveau2){
 62 	resetmdp();
 63 	if (pseudo.value==''){
 64 		$("#labpseudochmt2").text("/!\\ entrez votre pseudo.");
 65 		pseudo.focus();
 66 	}
 67 	
 68 	else if (ancien.value==''){
 69 		$("#labancienmdp").text("/!\\ ancien mot de passe.");
 70 		ancien.focus();
 71 	}
 72    
 73 	else if (nouveau.value==''){
 74 		$("#labnouveaumdp").text("/!\\ entrez votre nouveau mot de passe.");
 75 		nouveau.focus();
 76 	}
 77 	
 78 	else if (nouveau2.value==''){
 79 		$("#labconfirm").text("/!\\ confirmez mot de passe.");
 80 		nouveau2.focus();
 81 	}
 82 
 83 	else if (nouveau.value!=nouveau2.value){
 84 		$("#labnouveaumdp").text("/!\\ les mots de passe diffèrent.");
 85 		$("#labconfirm").text("/!\\");
 86 		nouveau.select();
 87 	}
 88 
 89 	else{
 90 		set_reception("EM_CMPT", "ACTPASS");
 91 		connexion_serveur();
 92 		socket_send("ACTPASS@"+pseudo.value+"@"+ancien.value+"@"+nouveau.value);
 93 	}
 94 }

134 /**
135  * Envoi une demande de connexion, après avoir vérifié la validité des variables.
136  * @author Paul	Mura
137  * @param {HTMLObject} psd Le pseudo de l'utilisateur.
138  * @param {HTMLObject} pass Son mot de passe.
139  */
140 function connexion(psd, pass){
141 resetcon();
142 	if (psd.value==''){
143 		$("#labpseudocon").text("/!\\ entrez votre pseudo.");
144 		psd.focus();
145 	}
146    
147 	else if (pass.value==''){
148 		$("#labpasscon").text("/!\\ entrez votre mot de passe.");
149 		pass.focus();
150 	}
151 	else{
152 		setpseudo(psd.value);
153 		pseudoC = psd.value;
154 		set_reception("EM_CMPT", "CONNECT");
155 		connexion_serveur();
156 		socket_send("CONNECT@"+psd.value+"@"+pass.value);
157 	}
158 }
