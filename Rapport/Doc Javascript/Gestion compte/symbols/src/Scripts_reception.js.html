 31 /**
 32  * Appelle la fonction de message correspondant au domaine au message reçu. 
 33  * @author Paul	Mura
 34  * @param {String} message Le message reçu depus le serveur.
 35  * @merberOf reception
 36  */
 37 function traiter_message(message){
 38 	console.log("RECEPTION :   "+message+"  "+domaine+" "+action);
 39 	if(message=="ERROR") alert("Erreur de syntaxe");
 40 	else if(domaine=="EM_CMPT"){
 41 		if(action=="GETINFO"){
 42 			getinfo(message);
 43 		}
 44 		else if(action=="CONNECT"){
 45 			connect(message);
 46 		}
 47 		else if(action=="ACTPASS"){
 48 			actpass(message);
 49 		}
 50 		else if(action=="ACTPSEUDO"){
 51 			actpseudo(message);
 52 		}
 53 		else if(action=="CREATCPT"){
 54 			creatcpt(message);
 55 		}
 56 	}
 57 	else if(domaine=="EM_PRT"){
 58 		if(action=="CREATEPARTIE"){
 59 			createpartie(message);
 60 		}
 61 		else if(action=="GETLISTEPARTIE"){
 62 			getlistepartie(message);
 63 		}
 64 		else if(action=="REJP"){
 65 			rejp(message);
 66 		}
 67 		else if(action=="GETPLAYERPARTY"){
 68 			getplayerparty(message);
 69 		}
 70 		else if(action=="EXITPARTIE"){	
 71 			exitpartie(message);
 72 		}
 73 		else if(action=="DEBUTPARTIE"){
 74 			debutpartie(message);
 75 		}
 76 		else if(action=="CHOIX"){
 77 			choix(message);
 78 		}
 79 	}
 80 	else if(domaine=="PARTIE"){
 81 		var act = message.split("@");
 82 
 83 		if(act[0]=="DEBUTPARTIE"){ alert("Début de la partie.");set_encours(true);}
 84 			
 85 		else if(act[0]=="LISTEJOUEURSPARTIE"){lister_joueurs(act);}
 86 		
 87 		else if(act[0]=="JETONJ") jetonj(act);
 88 		
 89 		else if(act[0]=="CARTEM") cartem(act);
 90 		
 91 		else if(act[0]=="CARTET") cartet(act);
 92 		
 93 		else if(act[0]=="JOUE") joue(act);
 94 		
 95 		else if(act[0]=="JETONT") jetont(act);
 96 		
 97 		else if(act[0]=="MONTREC") montrec(act);
 98 		
 99 		else if(act[0]=="GAGNANTT") gagnantt(act);
100 		
101 		else if(act[0]=="GAGNANTP") gagnantt(act);
102 		
103 		else if(act[0]=="PERDU") ;
104 		
105 		else if(act[0]=="AREUREADY"){socket_send("IAMREADY");}
106 		
107 		else if(act[0]=="JCHOIX") jchoix(act);
108 		
109 		else if(act[0]=="JOUEURJ") joueurj(act);
110 		
111 		else if(act[0]=="SETINFO") getinfo(message);
112 		
113 		else if(act[0]=="MESSAGE"){
114 			$("#prompt").append("\n"+act[1]+" dit : "+act[2]);
115 			$("#prompt").scrollTop = $("#prompt").scrollHeight;
116 		}
117 	}
118 }

123 /**
124  * Gère les messages reçus après une demande de connexion. Notifie l'utilisateur en cas d'erreur, appelle la fonction con() sinon.
125  * @author Paul	Mura
126  * @param {String} message
127  * @merberOf reception
128  */
129 function connect(message){
130 	if(message == 'CONNECTOK'){
131 		con();
132 	}
133 	else if(message == 'WPSEUDO'){
134 		resetcon();
135 		$("#labpseudocon").text("/!\\ pseudo inexistant");
136 		$("#pseudocon").select();
137 	}
138 	else if(message == 'WPASS'){
139 		resetcon();
140 		$("#labpasscon").text("/!\\ mot de passe incorrect");
141 		$("#passcon").select();
142 	}
143 	else if(message == 'PDC')
144 		alert("Vous êtes déjà connecté.");
145 		
146 	else alert("Serveur indisponnible. Veuillez réessayer ultérieurement.");
147 }
