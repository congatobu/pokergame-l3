  6 /**
  7  * Envoi une demande de création de partie, après avoir vérifié la validité des variables.
  8  * @author Paul	Mura
  9  * @param {HTMLObject} nom Le nom de la partie.
 10  * @param {HTMLObject} nb Le nombre de joueurs maximal dans la partie.
 11  */
 12 function creer_partie(nom, nb){
 13 	resetcreerprt();
 14 	var reg=new RegExp("^[2-8]{1}$","g");
 15 	
 16 	if (nom.value==''){
 17 		$("#labnompartie").text("/!\\ entrez un nom.");
 18 		nom.focus();
 19 	}
 20    
 21 	else if (nb.value==''){
 22 		$("#labnbjoueurs").text("/!\\ entrez un nombre de joueurs.");
 23 		nb.focus();
 24 	}
 25 	else if(!reg.test(nb.value)){
 26 		$("#labnbjoueurs").text("/!\\ entrez un nombre entre 2 et 8.");
 27 		nb.focus();
 28 	}
 29 	else{
 30 		set_nom_partie(nom);
 31 		set_createur(true);
 32 		set_reception("EM_PRT", "CREATEPARTIE");
 33 		socket_send("CREATEPARTIE@"+nom.value+"@"+nb.value);
 34 	}
 35 
 36 }
